之前说过await的原理是一个类似回调的东西，于是突发奇想能不能做一个符合如下条件的Awaiter:

  • 类似If语句可以判断是否执行后续逻辑
  • 类似Using在后续代码都结束时可以执行一个特定方法

于是写了类似如下代码

class DoOrSkip
{
      public bool IsCompleted => false;

      public bool condition;

      public void OnComplete(Action contin)
      {
            if(condition)
            {
                  contin+=()=>{
                        condition=true;
                        Debug.Log("Reset");
                  }
                  contin();
            }
      }
}


//Unity Code
class Test:MonoBehaviour
{
      async void Update()
      {
            await DoOrSkip();
            await Task.Delay(300);
            Debug.Log("Done")
      }
}

这段代码乍看感觉可以实现，但问题出在Task.Delay(300)这边导致回调并不阻塞，所以Reset将在await DoOrSkip()之后立马打印出来。

后面有想到使用using+await组合，通过await返回一个IDispose在using结尾调用，但问题出在await作用域和using不一样，await若跳过将会把后续代码都跳过，using只在中括号范围内。

所以后续找到了另一种方式——AsyncMethodBuilder用他来构建自己的await状态机
在Unity2020.3中的.net2中缺少一个所需的Attribute需要添加如下代码

#pragma warning disable CS1591 // Missing XML comment for publicly visible type or member
#pragma warning disable CS0436

namespace System.Runtime.CompilerServices
{
    internal sealed class AsyncMethodBuilderAttribute : Attribute
    {
        public Type BuilderType { get; }

        public AsyncMethodBuilderAttribute(Type builderType)
        {
            BuilderType = builderType;
        }
    }
}

然后构建自己的Builder

using System;
using System.Collections.Generic;
using System.Runtime.CompilerServices;
using UnityEngine;

class MyTaskMethodBuilder<T>
{
    public static MyTaskMethodBuilder<T> Create()
    {
        Debug.Log(".cto");
        return new MyTaskMethodBuilder<T>();
    }

    public void Start<TStateMachine>(ref TStateMachine stateMachine)
        where TStateMachine : IAsyncStateMachine
    {
        Debug.Log("start");
        stateMachine.MoveNext();
    }

    public void SetStateMachine(IAsyncStateMachine stateMachine)
    {
        
    }

    public void SetException(Exception exception)
    {
        
    }

    public void SetResult(T result)
    {
        Debug.Log("Result");
        while (act.Count>0)
        {
            act.Pop()?.Invoke();
        }
    }

    private Stack<Action> act = new Stack<Action>();

    public void AwaitOnCompleted<TAwaiter, TStateMachine>(
        ref TAwaiter awaiter, ref TStateMachine stateMachine)
        where TAwaiter : INotifyCompletion
        where TStateMachine : IAsyncStateMachine
    {
        if (awaiter is Awaiter<T> m)
        {
            act.Push(m.DoSelf);
        }
        Debug.Log("Next");
        stateMachine.MoveNext();
    }

    public void AwaitUnsafeOnCompleted<TAwaiter, TStateMachine>(
        ref TAwaiter awaiter, ref TStateMachine stateMachine)
        where TAwaiter : ICriticalNotifyCompletion
        where TStateMachine : IAsyncStateMachine
    {
        Debug.Log("NextUnsafe");
        stateMachine.MoveNext();
    }

    public MyTask<T> Task { get; }
}

在构造Task-Like对象

using System;
using System.Runtime.CompilerServices;
using UnityEngine;

[AsyncMethodBuilder(typeof(MyTaskMethodBuilder<>))]
class MyTask<T>
{
    private int i;
    public MyTask(int i)
    {
        this.i = i;
    }
    public Awaiter<T> GetAwaiter()
    {
        var a = new Awaiter<T>();
        a.i = i;
        return a;
    }
}

class Awaiter<T> : INotifyCompletion
{
    public bool IsCompleted { get; }
    public int i;
    public void DoSelf()
    {
        Debug.Log("Hahaha"+i);
    }

    public T GetResult()
    {
        return default;
    }

    public void OnCompleted(Action completion)
    {
        completion?.Invoke();
    }
}

这样就可以实现结束回调了，其余还有很多东西需要实现可以参考UniTask
