v2ray 透明代理案例

ip route add local 0.0.0.0/0 dev lo table 100
ip rule add fwmark 1 table 100 nftable 标记为1的走table 100的规则匹配路由

#!/usr/sbin/nft -f
# nft-set.sh
flush ruleset

define RESERVED_IP = {
    10.0.0.0/8,
    100.64.0.0/10,
    127.0.0.0/8,
    169.254.0.0/16,
    172.16.0.0/12,
    192.0.0.0/24,
    224.0.0.0/4,
    240.0.0.0/4,
    255.255.255.255/32
}

table ip xray {
        chain prerouting {
                type filter hook prerouting priority mangle; policy accept;
                #局域网直连
                ip daddr $RESERVED_IP return 
                #局域网非dns查询直连
                ip daddr 192.168.0.0/16 tcp dport != 53 return 
                ip daddr 192.168.0.0/16 udp dport != 53 return 
                #tproxy
                ip protocol tcp log flags all level debug prefix "nftables tproxy tcp to: " tproxy to 127.0.0.1:12345 meta mark set 1 
                ip protocol udp log flags all level debug prefix "nftables tproxy udp to: " tproxy to 127.0.0.1:12345 meta mark set 1
        }
        #tproxy不允许output重定向
        # 本地发起的链接都是从output开始的，访问时打个标记让他通过ip route设置的路由走到prerouting统一从prerouting处理流量
        chain output {
                type route hook output priority mangle; policy accept;
                ip daddr $RESERVED_IP return
                ip daddr 192.168.0.0/16 tcp dport != 53 return
                ip daddr 192.168.0.0/16 udp dport != 53 return
                #v2ray流量直连
                meta mark 255 log flags all level debug prefix "nftables v2ray to: " return 
                #通过route转发重新prerouting
                ip protocol tcp meta mark set 1 
                ip protocol udp meta mark set 1
        }
}
