  Unity在2019之后就可以使用UI ToolKit来制作，其制作理念和web前端基本一致，所有这里记录一下相关的内容。

基本结构

通过Create > UI Toolkit > Editor Window即可创建一个最基本的Editor窗口

using UnityEditor;  
using UnityEngine;  
using UnityEngine.UIElements;  
  
public class CustomEditor : EditorWindow  
{  
    [SerializeField]  
    private VisualTreeAsset m_VisualTreeAsset = default;  
  
    [MenuItem("Window/UI Toolkit/CustomEditor")]  
    public static void ShowExample()  
    {        CustomEditor wnd = GetWindow<CustomEditor>();  
        wnd.titleContent = new GUIContent("CustomEditor");  
    }  
    public void CreateGUI()  
    {        
        // Each editor window contains a root VisualElement object  
        VisualElement root = rootVisualElement;  
  
        // VisualElements objects can contain other VisualElement following a tree hierarchy.  
        VisualElement label = new Label("Hello World! From C#");  
        root.Add(label);  
  
        // Instantiate UXML  
        VisualElement labelFromUXML = m_VisualTreeAsset.Instantiate();  
        root.Add(labelFromUXML);  
    }}

修改鼠标样式

  在反编译查看了下ObjectField后发现了DragAndDrop.visualMode = DragAndDropVisualMode.Generic这一行代码，试了一下后发现可以用于修改拖拽时的鼠标样式

element.RegisterCallback<DragUpdatedEvent>(x =>  
{  
    DragAndDrop.visualMode = DragAndDropVisualMode.Generic;  
});

修改Sprite

修改资源时需要修改对应的Importer，AssetImporter.GetAtPath，有些修改项不在import中需要使用对应的ImporterSettings，如下面例子。


// 由于import setting一直处于选择状态的时候通过代码去改变会导致最后返回界面时再次弹出确实保存的框  
// 即选择状态下下面的东西没生效，需要点击一次保存  
// 取消选择则不会出现这种问题
var select = Selection.objects;  
Selection.objects = null;

// 批量修改Sprite的Pivot
_spritePaths.Select(x => AssetImporter.GetAtPath(x)).Where(x => x is TextureImporter).ForEach(x =>  
{  
    TextureImporter import = (TextureImporter)x;  
    var Texture = AssetDatabase.LoadAssetAtPath<Texture>(x.assetPath);  
    import.spritePixelsPerUnit = Texture.width;  
    var xy = root.Query<Vector2Field>().First().value;  
    // 创建ImporterSettings
    TextureImporterSettings texSettings = new TextureImporterSettings();
    // Copy之前的设置到ImporterSettings中  
    import.ReadTextureSettings(texSettings);
    // 修改项目  
    texSettings.spriteAlignment = (int)SpriteAlignment.Custom;  
    texSettings.spritePivot = xy;  
    // 将ImporterSettings写入import
    import.SetTextureSettings(texSettings);  
  
	import.SaveAndReimport();  
    AssetDatabase.WriteImportSettingsIfDirty(x.assetPath);  
    AssetDatabase.SaveAssetIfDirty(import);  
});  

Selection.objects = select;
