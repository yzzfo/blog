Openwrt中的开机自启.
Posted on 2023-06-11 :: Tags: Openwrt

在 OpenWRT 系统内 init 进程被 procd 取代，procd 作为父进程可以监控子进程的状态。一旦子进程退出后即可在某一个时刻尝试进行重启进程。它通过 init 脚本来将进程信息加入到 procd 的数据库中来管理进程启动，这是通过 ubus 总线调用来实现，可以防止进程的重复启动调用. 在 OpenWRT 系统内使用 procd 监控的有 uhttpd, netifd 等。在/etc/init.d/文件夹内的启动文件带有 USE_PROCD=1标志.

#!/bin/sh /etc/rc.common
# /etc/init.d
# Copyright (C) 2008 OpenWrt.org

START=98
STOP=98

USE_PROCD=1


start_service() {
        echo "start user service!"

        procd_open_instance npc

        procd_set_param respawn 3600 5 10

        procd_set_param command /root/npc -config /root/conf/npc.conf

        procd_close_instance
}


stop_service() {
        echo "stop user service!"
        killall npc
}


reload_service(){
        echo "reload user service!"
        stop
        start
}


restart() {
 stop
 start
}
