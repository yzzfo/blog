  单例是一种经常会见到的设计模式，但是单例对于新增接口以及之后的修改并不太友好(比如多个单例都实现同一个接口)，我们可以通过将其纳入到IOC中使得忽略该接口的实现细节(单例或者每次都实例化)直接获取接口对象，方便我们的管理。

环境

.NET 7.0
Sdk="Microsoft.NET.Sdk.Web"

关键API

构建阶段

  1. WebApplicationBuilder 管理服务和配置的构建者
  2. IServiceCollection 服务相关的注册
  3. ConfigurationManager 配置相关的注册

运行阶段

  1. IServiceProvider 用于获取服务的接口
  2. IOption IOptionX均用于获取配置
  3. IOptionSnapshot
  4. IOptionMotion

获取配置时采用IServiceProvider.GetService<IOptionX>方式获取

制作一个基于IOC的单例系统

  在ASP的mvc框架下，他的controller只需继承ControllerBase以及标记对于的Attribute就会在运行时自动注册上对应的路由，我们制作的单例系统也将实现类似的效果。

实现SinglontonAttribute

[AttributeUsage(AttributeTargets.Class)]
public class SinglontonAttribute : Attribute
{
    public bool RunningOnAwake = false;  //为true时，则在Setup时便会实例化该单例
    public int Order = 0;     // 实例化单例时的顺序
}

实现SinglontonAttribute的拓展方法

public static class SingletonsExt
{
    public static IServiceCollection AddSinglontons(this WebApplicationBuilder builder)
    {
        return builder.Services.AddSinglontons();
    }

    public static IServiceCollection AddSinglontons(this IServiceCollection server)
    {
        // 注册class为单例
        var method = typeof(ServiceCollectionServiceExtensions)
            .GetMethods(BindingFlags.Static | BindingFlags.Public)
            .First(x => x.Name == "AddSingleton" && x.GetParameters().Length == 1 && x.IsGenericMethod &&
                        x.GetGenericArguments().Length == 1);
        // 单例进行转发 AddSingleton<Interface>(x=>x.GetService(class))
        var methodForward = typeof(ServiceCollectionServiceExtensions)
            .GetMethods(BindingFlags.Static | BindingFlags.Public)
            .First(x => x.Name == "AddSingleton" && x.GetParameters().Length == 3 && !x.IsGenericMethod &&
                        x.GetParameters().Last().ParameterType.IsGenericType &&
                        x.GetParameters().Last().ParameterType.GetGenericTypeDefinition() == typeof(Func<,>));

        Assembly.GetExecutingAssembly().GetTypes().ForEach(type =>
        {
            if (type.GetAllAttribute<SinglontonAttribute>().Count > 0)
            {
                method.MakeGenericMethod(type)
                    .Invoke(null, new object[] { server });

                type.GetInterfaces().ForEach(i =>
                {
                    methodForward.Invoke(null,
                        new object[] { server, i, (IServiceProvider provide) => provide.GetService(type) });
                });
            }
        });
        return server;
    }

    public static WebApplication Setup(this WebApplication server)
    {
        Assembly assembly = Assembly.GetExecutingAssembly();
        List<(Type, int)> toAwakes = new List<(Type, int)>();
        assembly.GetTypes().ForEach(type =>
        {
            var attribute = type.GetCustomAttribute<SinglontonAttribute>(false);
            if (attribute != null && attribute.RunningOnAwake)
            {
                toAwakes.Add((type, attribute.Order));
            }
        });

        toAwakes.Sort((x, y) =>
        {
            if (x.Item2 != y.Item2)
            {
                return x.Item2 - y.Item2;
            }
            
            return String.CompareOrdinal(x.ToString(), y.ToString());
        });
        
        toAwakes.ForEach(x =>
        {
            server.Services.GetService(x.Item1);
        });

        return server;
    }
}

使用方法

// Program.cs
var builder = WebApplication.CreateBuilder(args);
builder.Services.AddSinglontons();  //注册单例
var app = builder.Build();
app.Setup();    //对特定单例实例化
app.Run();

实现一个简单的定时任务

// TimeServer.cs
// 一个维护定时任务的单例对象

public interface ITick
{
    public String ServerName => this.GetType().ToString();
    public int TickTime { get; }
    public Task Update();
}

[Singlonton(RunningOnAwake = true)]
public class TickServer
{
    private Dictionary<String, ITick> timeServers;
    private Dictionary<String, long> nextTime = new Dictionary<string, long>();
    // 这样在实例化的时候将自动把所有ITimeServer对象注入进来，而不用关系外界情况
    public TickServer(IEnumerable<ITick> timeServers)
    {
        this.timeServers = timeServers.ToDictionary((s) => { return s.ServerName; });
        foreach (var key in this.timeServers.Keys)
        {
            this.nextTime[key] = 0;
        }

        this.Running();
    }

    private async void Running()
    {
        await UniTask.SwitchToThreadPool();
        while (true)
        {
            foreach (var key in timeServers.Keys)
            {
                if (this.nextTime[key] > DateTime.Now.Ticks)
                {
                    continue;
                }

                this.nextTime[key] = DateTime.Now.AddSeconds(timeServers[key].TickTime).Ticks;
                await UniTask.SwitchToThreadPool();
                timeServers[key].Update();
            }

            await Task.Delay(100);
        }
    }
}

// TickServer实例化时也将自动实例化下面的类，并纳入时间管理

public class HelloTick : ITick    
{
    public int TickTime => 1;
    public Task Update()
    {
        Console.WrinteLine("Hello")

}
