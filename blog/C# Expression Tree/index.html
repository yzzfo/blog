在C#中Linq语句可以使得许多数据处理逻辑写起来十分方便，在Linq存在一种Expression的概念，用它可以动态构建许多方法用来增强逻辑处理能力。

利用表达式树映射对象属性

有时候会存在两个很相似的类，但他们却不隶属同一个父类或接口，这时候想进行相互赋值就可以使用表达式树来构建这种映射关系。

public class ExpressionMapper<TIn, TOut>
{
    static Func<TIn, TOut> e;
    static ExpressionMapper()
    {
        ParameterExpression parameterExpression = Expression.Parameter(typeof(TIn), "iv");

        List<MemberBinding> memberBindings = new List<MemberBinding>();

        typeof(TOut).GetProperties().ForEach((p) =>
        {
            MemberExpression member = Expression.Property(parameterExpression, typeof(TIn).GetProperty(p.Name));
            MemberBinding memberBinding = Expression.Bind(p, member);
            memberBindings.Add(memberBinding);
        });

        typeof(TOut).GetFields().ForEach((p) =>
        {
            MemberExpression member = Expression.Field(parameterExpression, typeof(TIn).GetField(p.Name));
            MemberBinding memberBinding = Expression.Bind(p, member);
            memberBindings.Add(memberBinding);
        });
        MemberInitExpression memberInitExp = Expression.MemberInit(Expression.New(typeof(TOut)), memberBindings.AsEnumerable());

        Expression<Func<TIn, TOut>> exp = Expression.Lambda<Func<TIn, TOut>>(memberInitExp, parameterExpression);

        e = exp.Compile();
    }

    public static TOut Trans(TIn i)
    {
        return e.Invoke(i);
    }
}

数据绑定

有时候知道两个数据应该是相互关联的，调整一个的时候另一个也要同时变化，这时候可以通过Lambda传递数据路径来构造表达式，进行数据关联。

    public class BindingValue<T> where T : IEquatable<T>
    {
        private Func<T> getFrom;
        private Func<T> getTo;

        private Action<T> setFrom;
        private Action<T> setTo;
        public Action<T, T> onValueChange;

        public T value
        {
            get => getFrom();
            set
            {
                if (!value.Equals(this.value))
                {
                    var ov = value;
                    setFrom(value);
                    setTo(value);
                    onValueChange?.Invoke(ov, value);
                }
            }
        }

        public BindingValue(Expression<Func<T>> from, Expression<Func<T>> to, T defalutValue = default)
        {
            getFrom = from.Compile();
            getTo = to.Compile();

            setFrom = BuildSet(from);
            setTo = BuildSet(to);
            value = defalutValue;
        }

        private Action<T> BuildSet(Expression<Func<T>> exp)
        {
            var param = Expression.Parameter(typeof(T));
            Expression left = exp.Body;
            Expression right = param;
            if (exp.Body is UnaryExpression uExp)     //处理强制转换
            {
                left = uExp.Operand as MemberExpression;
                right = Expression.Convert(param, left.Type);
            }

            var assign = Expression.Assign(left, right);
            var setExp = Expression.Lambda<Action<T>>(assign, param);
            return setExp.Compile();
        }
    }
