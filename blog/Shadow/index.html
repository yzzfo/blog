Unity的shader实现阴影的方法有多种，其中一种是使用Shadow Map技术。Shadow Map是一种记录光源视角下场景深度信息的纹理，可以用来判断物体是否在阴影区域。

如果想要渲染的物体接收阴影，就要在Shader中对阴影映射纹理进行采样，通过采样结果和光照结果相乘来产生阴影效果。Unity提供了一些内置的宏和函数来方便这个过程，例如SHADOW_COORDS, TRANSFER_SHADOW, SHADOW_ATTENUATION等。

如果想要渲染的物体产生阴影，就要在Shader中添加一个ShadowCaster Pass，用来告诉Unity如何生成Shadow Map。Unity也提供了一些内置的宏和函数来简化这个过程，例如SHADOW_CASTER_FRAGMENT等。

在普通Shader中实现接收Shadow

struct v2f_surf {
    float4 pos : SV_POSITION;
    float2 uv0 : TEXCOORD0;
    half3 normal : TEXCOORD1;
    SHADOW_COORDS(2)
};

v2f_surf vert (appdata_full v) {
    v2f_surf o;
    TRANSFER_SHADOW(o);
    return o;
}

half4 frag (v2f_surf i) : SV_Target {
    half attenuation = SHADOW_ATTENUATION(i);
    directColor = directColor * atten;
}

//AutoLight.cginc
#define SHADOW_COORDS(idx1) float4 _ShadowCoord : TEXCOORD##idx1;
#define TRANSFER_SHADOW(a) a._ShadowCoord = mul( _ShadowMatrix, mul( _Object2World, v.vertex ) );
#define SHADOW_ATTENUATION(a) unitySampleShadow(a._ShadowCoord)

阴影投射

在普通Shader中实现阴影投射

// Pass to render object as a shadow caster
Pass 
{
Name "ShadowCaster"
Tags { "LightMode" = "ShadowCaster" }

    CGPROGRAM
    #pragma vertex vert
    #pragma fragment frag
    #pragma target 2.0
    #pragma multi_compile_shadowcaster
    #pragma multi_compile_instancing // allow instanced shadow pass for most of the shaders
    #include "UnityCG.cginc"

    struct v2f {
        V2F_SHADOW_CASTER;
        UNITY_VERTEX_OUTPUT_STEREO
    };

    v2f vert(appdata_base v)
    {
        v2f o;
        UNITY_SETUP_INSTANCE_ID(v);
        UNITY_INITIALIZE_VERTEX_OUTPUT_STEREO(o);
        TRANSFER_SHADOW_CASTER_NORMALOFFSET(o)
        return o;
    }

    float4 frag(v2f i) : SV_Target
    {
        SHADOW_CASTER_FRAGMENT(i)
    }
    ENDCG
}
