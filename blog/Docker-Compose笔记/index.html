  1. Docker-Compose安装
    pip install docker-compose 或者 apt-get install docker-compose
  2. Docker-Compose使用
     #运行当前目录下的docker-compose.yaml
 docker-compose up -d

 #停止运行[并删除创建的volumes]
 docker-compose down [--volumes]

 #指定docker-compose.yaml
 -f /path/to/file

  4. compose中不常用选项的示例
     version: '3'
 services:
   app:
     build:
       context: ./dockerfile
       dockerfile: ./Dockerfile
     restart: always
     ports:
       - 8434:80
     cap_add:
       - ALL
     cap_drop:
       - NET_ADMIN
       - SYS_ADMIN
     privileged: true
     deploy:
       resources:
         limits:
           cpus: '0.50'
           memory: 50M
         reservations:
           cpus: '0.25'
           memory: 20M
