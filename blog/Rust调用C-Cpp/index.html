编写C代码

// library.h
#ifndef HELLOWORLD_LIBRARY_H
#define HELLOWORLD_LIBRARY_H

void hello(void);

#endif //HELLOWORLD_LIBRARY_H

// library.c
#include "library.h"

#include <stdio.h>

void hello(void) {
    printf("Hello, World!\n");
}

编写Rust绑定代码

# Cargo.toml
[build-dependencies]
bindgen = "0.69.4"
cc = "1.0"

// build.rs

// 让 `cc` crate 编译 C++ 代码
cc::Build::new()
    .cpp(false) // 表示这是 C++ 代码
    .file("/c/abc/library.c") // 你的 C++ 源文件
    .compile("helloworld");

// 生成绑定
let bindings = bindgen::Builder::default()
    .header("/c/abc/library.h") // 你的 C++ 头文件
    .generate()
    .expect("Unable to generate bindings");

// 将生成的绑定写入到 $OUT_DIR/bindings.rs
// $OUT_DIR一般为 target/release/build/xxxx/out
let out_path = PathBuf::from(env::var("OUT_DIR").unwrap());
bindings
    .write_to_file(out_path.join("bindings.rs"))
    .expect("Couldn't write bindings!");

若头文件很多，也可以写一个wrapper.h

// wrapper.h
#pragma once

#include "library.h"
#include "library2.h"
#include "library3.h"

// 生成绑定
let bindings = bindgen::Builder::default()
    .header("/c/abc/wrapper.h") // 你的 C++ 头文件
    .generate()
    .expect("Unable to generate bindings");

至此，Rust即可使用 C header中的方法了

关于.a和.o等文件

.o 文件

用途：这是目标文件，由编译器从 .c 或 .cpp 源文件生成。目标文件包含机器代码，但尚未链接成可执行文件或库。链接器会将多个 .o 文件和库文件链接在一起生成最终的可执行文件或库。
内容：编译后的机器代码和必要的符号信息。

.a 文件

用途：这是静态库文件，包含了一个或多个 .o 文件的集合。静态库在链接阶段被包含到最终的可执行文件中。与共享库不同，静态库在链接之后不需要随可执行文件一起分发。
内容：一个或多个 .o 文件的打包，通常用 ar 工具创建。

关于Wasm

在不使用C/C++的时候，rust可以直接通过wasm-pack来编译出wasm，但在依赖C/C++时且用到类似stdlib的时候则无法编译通过，因为wasm-pack没有默认的标准库，这时候需要使用emscripten 即 cargo build --target wasm32-unknown-emscripten --release

emscripten

在emscripten中很多指令和gcc类似，例如ar,dump等等，对应的指令为emar,emdump

以下是一个symbol去重的简易脚本，在特定情况下可以用于去重

import os
import subprocess

def extract_objects(archive, output_dir):
    subprocess.run(["emar.bat", "x", archive], cwd=output_dir)

def list_symbols(object_files):
    symbols = {}
    for obj in object_files:
        result = subprocess.run(["emnm.bat", obj], capture_output=True, text=True)
        for line in result.stdout.splitlines():
            parts = line.split()
            if len(parts) == 3 and parts[1] in ('T', 'D', 'B', 'C'):
                symbol = parts[2]
                if symbol in symbols:
                    print(symbol)
                    symbols[symbol].append(obj)
                else:
                    symbols[symbol] = [obj]
    return symbols

def remove_duplicates(symbols):
    to_keep = set()
    to_remove = set()
    for symbol, objs in symbols.items():
        if len(objs) > 1:
            for obj in objs[1:]:
                print("exist: "+objs[0] + " "+obj)
                to_remove.add(obj)
        to_keep.add(objs[0])
    return to_keep, to_remove

def create_combined_library(output_dir, to_keep):
    combined_archive = os.path.join(output_dir, "libCombined.a")
    if os.path.exists(combined_archive):
        os.remove(combined_archive)
    # subprocess.run(["emar.bat", "qc", combined_archive] + list(to_keep))
    subprocess.run(["emar.bat", "qc", combined_archive,"*.o"])
    subprocess.run(["emranlib.bat", combined_archive])
    return combined_archive

def main():
    libA = "./libkxttest.a"
    libB = "./libktx_read.a"
    libC = "./libktx_unity.a"
    libD = "./libobj_basisu_cbind.a"
    output_dir = "./"
    
    # os.makedirs(output_dir, exist_ok=True)
    extract_objects(libA, output_dir)
    extract_objects(libB, output_dir)
    extract_objects(libC, output_dir)
    extract_objects(libD, output_dir)
    # return
    object_files = [os.path.join(output_dir, f) for f in os.listdir(output_dir) if f.endswith(".o")]
    symbols = list_symbols(object_files)
    
    to_keep, to_remove = remove_duplicates(symbols)
    print("toremove")
    print(to_remove)
    for obj in to_remove:
        print(obj+"\n")
        # os.remove(obj)
    
    combined_archive = create_combined_library(output_dir, to_keep)
    print(f"Combined library created at: {combined_archive}")

if __name__ == "__main__":
    main()

下面是一个用于重命名的脚本，用于对C/C++进行重命名，会将文件夹/文件中特定字符串替换掉

import os

def replace_in_file(file_path, old_text, new_text):
    print(file_path)
    with open(file_path, 'r', encoding='utf-8', errors='ignore') as file:
        content = file.read()
    content = content.replace(old_text, new_text)
    with open(file_path, 'w', encoding='utf-8', errors='ignore') as file:
        file.write(content)

def replace_in_directory(directory, old_text, new_text):
    for dirpath, dirnames, filenames in os.walk(directory):
        for filename in filenames:
            if filename.endswith(('.cpp', '.h','inc','sln','txt')):
                # 替换文件内容
                file_path = os.path.join(dirpath, filename)
                replace_in_file(file_path, old_text, new_text)
                
                # 替换文件名
                new_filename = filename.replace(old_text, new_text)
                new_file_path = os.path.join(dirpath, new_filename)
                os.rename(file_path, new_file_path)
        
        # 替换文件夹名
        for dirname in dirnames:
            new_dirname = dirname.replace(old_text, new_text)
            old_dir_path = os.path.join(dirpath, dirname)
            new_dir_path = os.path.join(dirpath, new_dirname)
            os.rename(old_dir_path, new_dir_path)

# 使用示例
replace_in_directory('.', 'basisu', 'b2asisu')
replace_in_directory('.', 'basist', 'b2asist')
replace_in_directory('.', 'buminiz', 'b2uminiz')
