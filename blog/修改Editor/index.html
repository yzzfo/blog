   最近在制作技能的时候感觉到 Unity 内建的资源选择工具在选择自建类型的资源时十分的难受，所以尝试来修改一下 Unity 的编辑器。

整体方案参考 Ultimate Editor Enhancer 这个插件，其修改内建元素时采用的是 Harmony 这个插件，其原理与在雨中冒险mod制作 一文中的 BepInEx 差不多，都是通过 Hook 的方式进行修改。

修改ObjectField

入口

ObjectField 的绘制入口是 EditorGUI 下的 DoObjectField 函数，他已经在 ObjectFieldInterceptor 类中被 Hook 了，可以在 Prefix 中劫持点击事件，并打开自己对于的选择界面。

// 注册Hook
static ObjectFieldSelector()  
{  
    ObjectFieldInterceptor.OnGUIBefore += OnGUIBefore;  
}

// 执行Hook相关的逻辑
private static void OnGUIBefore(  
    Rect position,  
    Rect dropRect,  
    int id,  
    Object obj,  
    Object objBeingEdited,  
    Type objType,  
    Type additionalType,  
    SerializedProperty property,  
    object validator,  
    bool allowSceneObjects,  
    GUIStyle style)  
{  
    if (!Prefs.objectFieldSelector) return;  
    if (property != null) return;  

	// 检查事件
    Event e = Event.current;  
    if (e.type == EventType.MouseUp && blockMouseUp)  
    {        blockMouseUp = false;  
        e.Use();  // 将事件消耗掉，防止他触发原本的ObjectField点击事件
        return;  
    }    if (e.type != EventType.MouseDown) return;  
  
    if (e.button == 1)  
    {        Rect rect = new Rect(position);  
        rect.xMin = rect.xMax - 16;  
        if (!rect.Contains(e.mousePosition)) return;  
    }    else if(e.button == 0)  
    {        Rect rect = new Rect(position);  
        rect.xMin = rect.xMax - 36;  
        rect.xMax -= 16;  
        if (!rect.Contains(e.mousePosition)) return;  
    }    else  
    {  
        return;  
    }  
  
    Type type = objType;  
  
    blockMouseUp = true;  
    e.Use();  
  
    position.xMin += EditorGUIUtility.labelWidth;  

	// 打开界面
    OpenAssetSelector(id, type);  
}
