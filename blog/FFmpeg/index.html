FastStart

  最近购入了一台Servarica的大盘鸡，在使用nplayer在线播放动漫时发现首帧加载的速度有点慢，于是研究了一下相关的解决方法，最终选择使用FFmpeg对MP4进行faststart处理，对于这种没有GPU的vps来说也是最快捷的方式了，配合unmanci就可以实现自动化faststart。

#!/bin/sh
#用于unmanic的转换脚本
#$1 源文件
#$2 输出文件
echo "-------------------------------------------------------------------------------------------------------------------------------------"
echo "$1" | grep -e "\.mp4" >/dev/null
if [ $? = "0" ]; then #mp4
    ffprobe -v debug "$1" 2>&1 | grep "0 seeks"  #mp4是否已经faststart了
    if [ $? = "0" ]; then
        echo "Is moov skip"
        exit 0
    fi
    ffmpeg -nostdin -i "$1" -movflags faststart -acodec copy -vcodec copy -threads 10 -preset ultrafast "$2" -y
    if [ $? = "0" ]; then
        echo "--------------------------------------------------------MP4 Move--------------------------------------------------------------------------"
        mv "$2" "$1"
        exit 0
    fi
else #mkv
    ffmpeg -nostdin -i "$1" -c copy -reserve_index_space 50k "$2" # 50kbytes is the suggested size for 1 hour of media.   # -nostdin 必须加上，否则会导致ffmpeg读取的数据变奇怪  https://superuser.com/questions/1088032/while-read-loop-through-lines-in-text-file-losing-characters-in-bash-script-i
    if [ $? = "0" ]; then
        echo "--------------------------------------------------------MKV Move--------------------------------------------------------------------------"
        mv "$2" "$1"
        exit 0
    fi
fi
exit 1
