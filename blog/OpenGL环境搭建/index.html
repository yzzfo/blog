环境搭建

安装

  1. 下载cmake并加入环境变量中
  2. 安装LLVM，用于获得clangd使得helix中的支持cpp代码提示

项目配置

Root
  - third
    - glfw
    - glew
  - cmake-ninja
  - main.cpp
  - CMakeLists.txt
  - compile_commands.json
  1. 下载glfw和glew，解压至third文件夹中
  2. 编写CMakeLists.txt文件
cmake_minimum_required(VERSION 3.22)
project(GLSLStudy)

set(CMAKE_CXX_STANDARD 14)
#导出compile_commands.json，用于clangd进行代码提示
set(CMAKE_EXPORT_COMPILE_COMMANDS ON)


include_directories(${PROJECT_SOURCE_DIR}/third/glew-2.1.0/include)
include_directories(${PROJECT_SOURCE_DIR}/third/glfw-3.3.8/include)


set(GLEW_LIB ${PROJECT_SOURCE_DIR}/third/glew-2.1.0/lib/Release/x64)
set(GLFW_LIB ${PROJECT_SOURCE_DIR}/third/glfw-3.3.8/lib-vc2022)

#设置lib搜索路径
#需要位于add_executable之前执行
link_directories(
        ${GLEW_LIB}
        ${GLFW_LIB}
)


add_executable(GLSLStudy main.cpp)
find_package(OpenGL REQUIRED)
#target_link_libraries(GLSLStudy OpenGL::GL ${GLEW_LIB}/glew32s.lib ${GLFW_LIB}/glfw3.lib)
target_link_libraries(GLSLStudy OpenGL::GL glew32s glfw3)
  1. CMake

  2. Hello World
#include <stdio.h>

#define GLEW_STATIC

#include <GL/glew.h>
#include <GLFW/glfw3.h>
#include <iostream>

int main() {
    std::cout << "Hello, World!" << std::endl;
    glfwInit();
    glfwWindowHint(GLFW_CONTEXT_VERSION_MAJOR, 3);
    glfwWindowHint(GLFW_CONTEXT_VERSION_MINOR, 3);
    glfwWindowHint(GLFW_OPENGL_PROFILE, GLFW_OPENGL_CORE_PROFILE);

    GLFWwindow *window = glfwCreateWindow(800, 600, "Yahah", NULL, NULL);

    if (window == NULL) {
        std::cout << "Failed" << std::endl;
        glfwTerminate();
        return -1;
    }

    glfwMakeContextCurrent(window);

    // glew用于自动配置opengl驱动相关内容
    glewExperimental = true;
    if (glewInit() != GLEW_OK) {
        printf("Init GLEW failed\n");
        glfwTerminate();
        return -1;
    }

    while (!glfwWindowShouldClose(window)) {
        glClearColor(0.2f, 0.3f, 0.2f, 1.0f);
        glClear(GL_COLOR_BUFFER_BIT);
    }
    return 0;
}

  1. 其他
    也可以通过其他方式替代执行vcvars64.bat，即通过使用Develop Command Prompt for VS 2022这个终端，在这个终端中关于vs的环境变量全部都已经加载好了，直接使用即可。

配置Helix

新建.helix文件夹，作用类似于.vscoed文件夹

配置LSP

覆盖默认cpp的lsp配置，其中--limit-results=0用于禁用返回结果的数量限制(因为GL开头的元素很多，导致超出了最大数量限制，进而导致代码提示不全)。 --fallback-style=Microsoft用于配置代码格式化为Microsoft的规则。

#languages.toml
[[language]]
name = "cpp"

language-server = { command = "clangd", args=[
#    "--compile-commands-dir=compile_commands_directory", #compile_commands.json的文件夹
#    "--completion-limit", #https://reviews.llvm.org/D42669
    "--limit-results=0", #https://github.com/helix-editor/helix/issues/2505
    "--background-index",
    # "--pretty",
    # "--log=verbose",
    "--fallback-style=Microsoft"  #https://clang.llvm.org/docs/ClangFormatStyleOptions.html#basedonstyle
] }
#language-server.timeout = 30 # in seconds, 20 is the default
#indent = {tab-width = 4,unit = " " }
