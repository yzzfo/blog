在Unity中直接载入png,jpg之类图片的时候对内存的压力是很大的，在Android和IOS可以使用astc格式，win下则可以使用dxt格式格式进行内存压缩，在使用3000x8000图片的时候能将内存从100M降低到10M左右。

ASTC

从https://github.com/ARM-software/astc-encoder下载相应软件后执行类似./astcenc-avx2.exe -cl ./测试.png test.astc 8x8 -medium -yflip指令即可生成astc图片，-yflip用于翻转图片以适应Unity。
ASTC的前16字节用于描述该图片的信息

struct astc_header
{
    uint8_t magic[4];
    uint8_t block_x;
    uint8_t block_y;
    uint8_t block_z;
    uint8_t dim_x[3];
    uint8_t dim_y[3];
    uint8_t dim_z[3];
};

通过dim_x和dim_y即可获得图片尺寸(小端字节序)
width = dim_x[0] + (dim_x[1] << 8) + (dim_x[2] << 16);
height = dim_y[0] + (dim_y[1] << 8) + (dim_y[2] << 16);

在Unity中载入ASTC需要剔除文件头(16字节Loading ASTC texture from file)

sbyte[] sbytes = (sbyte[])(Array)img_bytes;
width = sbytes[7] + (sbytes[8] << 8) + (sbytes[9] << 16);  
height = sbytes[10] + (sbytes[11] << 8) + (sbytes[12] << 16);  
var tex = new Texture2D(width,height,ASTC_4x4);
tex.LoadRawTextureData(img_bytes.Skip(6).ToArray());
tex.Apply();

DXT

下载https://github.com/FrozenStormInteractive/Crunch2，通过如下命令crunch -file blah.tga -fileformat dds -dxt1 -mipmode none生成dxt文件

// https://discussions.unity.com/t/can-you-load-dds-textures-during-runtime/84192/2
public static Texture2D LoadTextureDXT(byte[] ddsBytes, TextureFormat textureFormat)
{
    if (textureFormat != TextureFormat.DXT1 && textureFormat != TextureFormat.DXT5)
        throw new Exception("Invalid TextureFormat. Only DXT1 and DXT5 formats are supported by this method.");

    byte ddsSizeCheck = ddsBytes[4];
    if (ddsSizeCheck != 124)
        throw new Exception("Invalid DDS DXTn texture. Unable to read");  //this header byte should be 124 for DDS image files

    int height = ddsBytes[13] * 256 + ddsBytes[12];
    int width = ddsBytes[17] * 256 + ddsBytes[16];

    int DDS_HEADER_SIZE = 128;
    byte[] dxtBytes = new byte[ddsBytes.Length - DDS_HEADER_SIZE];
    Buffer.BlockCopy(ddsBytes, DDS_HEADER_SIZE, dxtBytes, 0, ddsBytes.Length - DDS_HEADER_SIZE);

    Texture2D texture = new Texture2D(width, height, textureFormat, false);
    texture.LoadRawTextureData(dxtBytes);
    texture.Apply();

    return (texture);
}
