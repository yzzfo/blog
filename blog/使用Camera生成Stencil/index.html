  有时候我们想让某个Layer上的所有物体都作为Stencil时就需要使用一个新的Camera用以渲染出对应的Stencil。

环境

  1. Unity2020
  2. Standard Render Pipline

相关API

  1. Camera.Render 手动调用相机的渲染，该相机必需是disable的
  2. Camera.SetReplacementShader 让该相机渲染的物品都使用指定Shader
  3. Camera.ResetReplacementShader 清除SetReplace的设置
  4. Camera.SetTargetBuffers 设置该相机渲染结果存到哪
  5. Camera.CopyFrom 复制另一个相机的设置
  6. GL.Clear 清除RenderTexture.active中的内容
  7. RenderTexture.DiscardContents 清理RenderTexture中的内容

思路

  1. 将用于渲染Stencil的Camera挂载于Main Camera之下
  2. Stencil Camera使用CopyFrom复制Main Camera的设置
  3. Stencil Camera的clearFlags设置为CameraClearFlags.Nothing
  4. Stencil Camera的cullingMask设置为对应的LayerMask
  5. 在Main Camera中的OnPostRender时调用Stencil Camera的Render方法生成Stencil，之所以在OnPoseRender时调用是因为此时可以获取到正确的Depth Buffer。

大体代码如下

public class MainRender : MonoBehaviour
{
    private static MainRender _Ins;
    //TODO Singlontan Logic
    // ...

    private RenderTexture renderTexture;
    private RenderTexture toRender;
    private void Awake()
    {
        camera = this.GetComponent<Camera>();
        //之所以创建一个Texture是为了可以在后边拿到Depth Buffer
        renderTexture = new RenderTexture(Screen.width, Screen.height, 24);
        camera.targetTexture = renderTexture;
        renderTexture.name = "MainRenderTexture";
    }

    private void Update()
    {
        ClearOutRenderTexture(this.renderTexture);
        camera.targetTexture = renderTexture;
    }

    private void OnPostRender()
    {
        //可以替换Shader提升性能
        //不替换的好处在于可以应对顶点动画以及透明度等情况
        // StencilRender.Ins.SetRenderShader("shader","tag");   
        toRender = StencilRender.Ins.RTImage(renderTexture);
        camera.targetTexture = null;    //设置后下方的dest就是输出到屏幕
    }

    private void OnRenderImage(RenderTexture src, RenderTexture dest)
    {
        Graphics.Blit(toRender,dest);   //将模板结果输出到屏幕
    }

    private void OnDestroy()
    {
        renderTexture.Release();
    }

    public void ClearOutRenderTexture(RenderTexture renderTexture)
    {
        RenderTexture rt = RenderTexture.active;
        RenderTexture.active = renderTexture;
        GL.Clear(true, true, Color.clear);
        RenderTexture.active = rt;
    }
}

public class StencilRender : MonoBehaviour
{
    //TODO Singlontan Logic
    // ...
    public RenderTexture renderTexture;
    private Material thresholdMat;
    private RenderTexture halfBufferA;
    void Awake()
    {
        Shader threshold = Shader.Find("Hidden/Include/ColorThreshold");
        this.thresholdMat = new Material(threshold);
        this.thresholdMat.SetColor(Threshold, new Color(0.1f, 0.1f, 0.1f, 0.1f));
        this.halfBufferA = RenderTexture.GetTemporary(Screen.width/2, Screen.height/2, 0, RenderTextureFormat.ARGB32);
    }
    public RenderTexture RTImage(RenderTexture src = null)
    {
        ClearOutRenderTexture(renderTexture);
        renderCamera.SetTargetBuffers(renderTexture.colorBuffer,
            src == null ? renderTexture.depthBuffer : src.depthBuffer);
        renderCamera.Render();
        Graphics.Blit(renderTexture, halfBufferA, thresholdMat);    //将图像二值化
        return halfBufferA;
    }

    public void SetRenderShader(string shader = null, string replacementTag = "")
    {
        Shader toSet = null;
        if (!string.IsNullOrEmpty(shader))
        {
            toSet = Shader.Find(shader);
        }

        renderCamera.ResetReplacementShader();
        renderCamera.SetReplacementShader(toSet, replacementTag);
    }

    public void ClearOutRenderTexture(RenderTexture renderTexture)
    {
        RenderTexture rt = RenderTexture.active;
        RenderTexture.active = renderTexture;
        GL.Clear(true, true, Color.clear);
        RenderTexture.active = rt;
    }

    private void OnDestroy()
    {
        renderTexture.ReleaseTemporary();
    }
}

二值化图片Shader

Shader "Hidden/Include/ColorThreshold"
{
	Properties
	{
		_MainTex ("Texture", 2D) = "white" {}
		_Threshold ("Threshold", Color) = (0,0,0,0)
	}
	SubShader
	{
		// No culling or depth
		Cull Off ZWrite Off ZTest Always

		Pass
		{
			CGPROGRAM
			#pragma vertex vert
			#pragma fragment frag
			
			#include "UnityCG.cginc"

			struct appdata
			{
				float4 vertex : POSITION;
				float2 uv : TEXCOORD0;
			};

			struct v2f
			{
				float2 uv : TEXCOORD0;
				float4 vertex : SV_POSITION;
			};

			v2f vert (appdata v)
			{
				v2f o;
				o.vertex = UnityObjectToClipPos(v.vertex);
				o.uv = v.uv;
				return o;
			}
			
			sampler2D _MainTex;
			float4 _Threshold;

			float4 frag (v2f i) : SV_Target
			{
				float4 col = tex2D(_MainTex, i.uv);
				col = step(_Threshold,col);
				// return float4(1,1,1,1);
				return col;
			}
			ENDCG
		}
	}
}

关于new和GetTemporary

这两个API均可以获得到一个RenderTexture，区别在于GetTemporary Unity 内部保留了一个临时渲染纹理池，因此对 GetTemporary 的调用通常只返回一个已经创建的纹理（如果大小和格式匹配）。这些临时渲染纹理在几帧未使用时实际上会被破坏。
