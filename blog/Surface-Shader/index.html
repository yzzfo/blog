  Surface Shader是Unity进行抽象过的一层Shader，有时候想基于Unity默认Shader的基础上加一些特效时使用Suface Shader在合适不过了，创建默认的Surf Shader时便以及实现了PBR相关的功能。

使用方法

编译指令

# pragma surface surfaceFunction lightModel [optionalparams] 类似透明度测试之类的东西需要在optionalparams处声明

Input

只能使用Unity提供的相关变量，诸如worldPos，screenPos等

顶点动画

在optionalparams中加入vertex:VertexFunction，并编写对应函数即可

#pragma surface surf Lambert vertex:vert 
struct Input 
{ 
	float2 uv_MainTex; 
}; 
float _Amount;
void vert (inout appdata_full v) 
{ 
	v.vertex.xyz += v.normal * _Amount;
}

阴影

在默认的模板情况下，我们直接使用alphatest剔除像素后，该物体所投射的阴影依旧是完整的，这时需要在optionalparams加入addshadow指令

一个Dissovle的例子

Shader "Custom/NewSurfaceShader"  
{  
    Properties  
    {  
        _Color ("Color", Color) = (1,1,1,1)  
        [HDR] _HColor ("HColor", Color) = (1,1,1,1)  
        _MainTex ("Albedo (RGB)", 2D) = "white" {}  
        _Noise ("Noise (RGB)", 2D) = "white" {}  
        _Glossiness ("Smoothness", Range(0,1)) = 0.5  
        _Metallic ("Metallic", Range(0,1)) = 0.0  
        _Height("Height",float) = 0  
        _Clip("Clip",float)=0.01  
    }  
    SubShader  
    {  
        Tags  
        {  
            "RenderType"="Opaque"  
        }  
        LOD 200  
  
        CGPROGRAM  
        // Physically based Standard lighting model, and enable shadows on all light types  
        #pragma surface surf Standard fullforwardshadows alphatest:_Clip addshadow
  
        // Use shader model 3.0 target, to get nicer looking lighting  
        #pragma target 3.0  
  
        sampler2D _MainTex;  
        sampler2D _Noise;  
  
        struct Input  
        {  
            float2 uv_MainTex;  
            float2 uv_Noise;  
            float3 worldPos;  
        };  
        half _Glossiness;  
        half _Metallic;  
        fixed4 _Color;  
        float4 _HColor;  
        float _Height;  
  
        // Add instancing support for this shader. You need to check 'Enable Instancing' on materials that use the shader.  
        // See https://docs.unity3d.com/Manual/GPUInstancing.html for more information about instancing.        // #pragma instancing_options assumeuniformscaling        UNITY_INSTANCING_BUFFER_START(Props)  
        // put more per-instance properties here  
        UNITY_INSTANCING_BUFFER_END(Props)  
  
        void surf(Input IN, inout SurfaceOutputStandard o)  
        {            // Albedo comes from a texture tinted by color  
            fixed4 c = tex2D(_MainTex, IN.uv_MainTex) * _Color;  
            o.Albedo = c.rgb;  
            // Metallic and smoothness come from slider variables  
            o.Metallic = _Metallic;  
            o.Smoothness = _Glossiness;  
  
            float3 move = float3(UNITY_MATRIX_M[0].w, UNITY_MATRIX_M[1].w, UNITY_MATRIX_M[2].w);  
            float height = (IN.worldPos - move).y;  
            float alpha = 1 - smoothstep(_Height, _Height + 0.1, height); // 0-1  
            o.Alpha = alpha;  
  
            half ifNotGrand = step(0.99, alpha); //0 则是渐变区域, 1是非渐变  
            half ifGrand = (1 - ifNotGrand); //1 则是渐变，0则是非渐变  
  
  
            o.Alpha = ifNotGrand * o.Alpha + ifGrand * o.Alpha * step(0.1, tex2D(_Noise, IN.uv_Noise).r);  
            o.Albedo = ifNotGrand * o.Albedo + ifGrand * (alpha * o.Albedo + (_HColor * (1 - alpha)));  
        }        ENDCG  
    }  
    FallBack "Diffuse"  
}
