画函数图像

//f(x) = x^3
float Plot1(float2 st)
{
	return smoothstep(-0.01,0.01,pow(st.x,3)-st.y)*(1-smoothstep(-0.01,0.01,pow(st.x,3)-st.y));
}

float Plot2(float2 st,float f)
{
	return smoothstep(f-0.01,f,st.y)-smoothstep(f,f+0.01,st.y);
	//smoothstep可以通过向左平移f单位得到如下
	//return smoothstep(-0.01,0,st.y-f)-smoothstep(0,0.01,st.y-f);
}

两种不同方式下的smooth函数区别

两者都做到了 （-0.01，0.01）区间范围内的显示，但Plot1的中单个Smooth长度为0.02，而Plot2中单个Smooth长度为0.01这是两者的一种差别。即Plot1是基于原本区间进行的变化操作，而Plot2则是通过两个不同区间的smooth合并而成的操作。

Sobel边缘检测

// Sample the source texture  
fixed4 col = tex2D(_MainTex, i.uv);  
  
// Calculate the texel size  
float dx = _MainTex_TexelSize.x;  
float dy = _MainTex_TexelSize.y;  
  
// Apply Sobel operator  
float3x3 sobelX = float3x3(-1, 0, 1,  
                           -2, 0, 2,  
                           -1, 0, 1);  
  
float3x3 sobelY = float3x3(-1, -2, -1,  
                           0, 0, 0,  
                           1, 2, 1);  
  
float gx = 0;  
float gy = 0;  
  
for (int y = -1; y <= 1; y++)  
{  
    for (int x = -1; x <= 1; x++)  
    {        fixed4 c = tex2D(_MainTex, i.uv + float2(x * dx, y * dy));  
        gx += Gray(c) * sobelX[x + 1][y + 1];  
        gy += Gray(c) * sobelY[x + 1][y + 1];  
    }}
}

float gmag = sqrt(gx * gx + gy * gy);

# 大于0.5的时候可以认为是边缘了
gmag = smoothstep(0.5 - _OutlineWidth * 0.5, 0.5 + _OutlineWidth * 0.5, gmag);  
  
// Return the outline color or the source color  
return lerp(col, _OutlineColor, gmag);

Shader画正多边形

float NPoly(int n,float r,float offset,float st)
{
    float perAngle = TWO_PI/n;
    float angle = atan2(st.y,st.x)+offset;
    angle = frac(angle/perAngle)*perAngle;
    float t = r*cos(perAngle/2)/cos(perAngle/2-angle);
    return 1-step(t,distance(float2(0,0),st));
}

float NPloy2(int n,float r,float offset,float st)
{
    float perAngle = TWO_PI/n;
    float angle = atan2(st.y,st.x)+offset;
    return step(cos(floor(.5+angle/perAngle)*angle-a),r);
}
