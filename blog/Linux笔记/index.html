例子

  1. 日志监控
     #!/usr/bin/bash

 color() {
     while read -r foo
     do
         bc=40
         if [ "$1" = "server log ->" ];then
             bc=44
         fi

         match=`echo $foo | grep -o "\[[A-Z]\]"`
         if [ $? = "0" ];then
             case $match in
             "[D]") echo -e "\033[$bc;37;4m$1 $foo\033[0m";;
             "[W]") echo -e "\033[$bc;33;4m$1 $foo\033[0m";;
             "[E]") echo -e "\033[$bc;31;4m$1 $foo\033[0m";;
             *)     echo -e "\033[$bc;37;4m$1 $foo\033[0m";;
             esac
             continue
         fi

         echo -e "\033[$bc;37;4m$1 $foo\033[0m"
     done
 }

 client () { 
   tail -q -n 0 -F ../../designer/trunk/client/logs/Client*.log | color "client log ->"
 }

 server () { 
   tail -q -n 0 -F ../../designer/trunk/logs/LogicS*.log | color "server log ->"
 }

 if [ -z "$1" ]; then
     cat <(client & server)
 else
     cat <(client & server) | grep -E "\[$1\]"
 fi

Shell 下脚本变量问题

file=$1  
name=$(basename $file .go)  
os=$2  
arch=$3  
ldflags='-ldflags "-s -w"'
suffix=""  
if [[ $os == "windows" ]]; then  
suffix=".exe"  
fi  
  
# https://stackoverflow.com/questions/39583838/bash-pass-variable-as-argument-with-quotes  
# 问题类似Docekr的CMD之类的原因，$ldflags被一整块识别之类的  
# 会出 "\"-s" 之类的报错
# GOOS=$os GOARCH=$arch go build $ldflags -o "$outputDir/$os/$arch/$name$suffix" $file  
  
cmd="GOOS=$os GOARCH=$arch go build $ldflags -o $outputDir/$os/$arch/$name$suffix $file"  
eval "$cmd"

mv文件

在使用rclone的时候，直接使用mv可能会存在无法ctrl+c进行中断，这时候使用rsync可以很好的解决这些问题，同时还可以显示进度

rsync -av --progress --remove-source-files ./sources/* ./target/path
