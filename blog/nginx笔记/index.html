添加Header

# 设置解码类型为gzip
# 将原本的Content-type设置为application/wasm
location ~ .+\.wasm.gz$ {
 root /www/game;
add_header Content-Encoding gzip;
add_header Content-type  application/wasm always;
}

# 设置其他.gz结尾的解码类型为gzip
location ~ .+\.gz$ {
 root /www/game;
add_header Content-Encoding gzip;
}

# index目录
location / {
 root /www/game;
}

设置反代超时

由于 cloudflare 有个 100s 的强制 timeout，所以若使用了 cloudflare 且设置的 timeout 大于 100s，还是会在 100s 就断开。

proxy_read_timeout 1d;
proxy_send_timeout 1d;
proxy_connect_timeout 1d;

正则表达式

在 Nginx 配置文件中，可以使用正则表达式来匹配 URL 或其他文本。以下是 Nginx 中正则表达式的使用示例：

  1. 简单的正则表达式匹配

使用 ~ 运算符可以在 Nginx 中使用正则表达式匹配 URL，例如：

location ~ /blog/\d+ {
    # 匹配 URL 为 /blog/ 后面跟一或多个数字的请求
    # 其他指令
}

在上面的示例中，正则表达式 \d+ 匹配一个或多个数字，匹配的 URL 将被重定向到该位置。

  1. 匹配扩展名

使用 ~ 运算符和 $ 字符可以在 Nginx 中使用正则表达式匹配文件扩展名，例如：

location ~ \.jpg$ {
    # 匹配所有以 .jpg 结尾的 URL
    # 其他指令
}

在上面的示例中，正则表达式 \. 匹配任何字符，. 字符表示点号，jpg$ 匹配 URL 末尾的 .jpg。

  1. 命名捕获

使用 location 指令和正则表达式的命名捕获组可以在 Nginx 中捕获 URL 中的特定部分，并将其传递给后端应用程序，例如：

location ~ ^/user/(?<username>[a-zA-Z0-9_-]+)$ {
    # 匹配 URL 为 /user/ 后面跟字母、数字、下划线或破折号的请求，将匹配到的用户名传递给后端应用程序
    # 其他指令
    proxy_pass http://backend/$username;
}

在上面的示例中，(?<username>[a-zA-Z0-9_-]+) 是一个命名捕获组，它匹配 URL 中 /user/ 后面的任何字母、数字、下划线或破折号，并将其保存到名为 username 的变量中。然后，可以使用 $username 变量将该值传递给后端应用程序。

需要注意的是，在使用正则表达式时，应该谨慎地考虑性能和安全性，并避免在表达式中使用过于复杂的模式。

localtion和if中正则表达式的不同

在localtion中 / 充当着路径的意思，所以会出现如下问题

#跨域匹配 $1为协议 $2为 /路径
#/cors/https://github.com/   $1 https $2 /github.com/
location ~* ^/cors/(.*):/(.*)$ {
}

#若改成 location ~* ^/cors/(.*)://(.*)$ 则会无法匹配上 /cors/https://github.com/

url decode的问题

在location中匹配的表达式将会被自动url decode 例如/cors/https://example.com/A%2FB 这将导致 $2 得到 /example.com/A/B 若依照这个值去proxy_pass显然是错误的
为了解决这个问题，需要使用if进行正则匹配，这时不会进行decode

if ($request_uri ~* ^/cors/(.*)$) {
    proxy_pass $1; #此时$1 为https://example.com/A%2FB
}

Cors Anywhere


#user  nobody;
worker_processes 1;

#error_log  logs/error.log;
#error_log  logs/error.log  notice;
#error_log  logs/error.log  info;

#pid        logs/nginx.pid;
events {
    worker_connections 1024;
}


http {
    include mime.types;
    default_type application/octet-stream;
    error_log logs/debug.log debug;
    log_format main '$remote_addr - $remote_user [$time_local] "$request" '
    '$status $body_bytes_sent "$http_referer" '
    '"$http_user_agent" "$http_x_forwarded_for" "$proxy_host"';

    access_log logs/access.log main;
    sendfile on;
    #tcp_nopush     on;

    #keepalive_timeout  0;
    keepalive_timeout 65;
    resolver 1.1.1.1;

    #gzip  on;

    server {
        listen 80;
        server_name localhost;

        #charset koi8-r;

        #access_log  logs/host.access.log  main;
        location ~* ^/cors/(.*):/(.*)$ {
            proxy_ssl_server_name on;
            #proxy_ssl_session_reuse off;
            proxy_buffer_size 64k;
            proxy_buffers 32 32k;
            proxy_busy_buffers_size 128k;
            proxy_connect_timeout 60;
            proxy_read_timeout 120;
            proxy_send_timeout 60;

            proxy_pass_request_headers on;
            proxy_set_header referer $1://$proxy_host;

            #proxy_redirect off;
            #$request_uri ~* ^/cors/(.*)$
            # 由于重新正则匹配之前的匹配值会被覆盖，即便?<name>也会被覆盖，例如$1路由别名sche,但在该if的regex执行后，sche也变成这次的$1了
            #且add_header始终位于proxy之后执行，所以继续使用同路由一样的匹配，这样就能保证$1和$2为期待值。
            if ($request_uri ~* ^/cors/(.*):/(.*)$) {
                proxy_pass $1:/$2;
            }
            proxy_redirect .* /;
            #proxy_redirect off;
            add_header "Set-Cookie" "CORS=$1://$proxy_host; HttpOnly; Secure;Path=/";
            add_header 'Access-Control-Allow-Origin' "$http_origin";
            add_header 'Access-Control-Allow-Methods' 'GET, POST, OPTIONS';
            add_header 'Access-Control-Allow-Headers' 'Origin, X-Requested-With, Content-Type, Accept';
        }

        location ~* ^/(.*)$ {
            proxy_ssl_server_name on;
            #proxy_ssl_session_reuse off;
            proxy_buffer_size 64k;
            proxy_buffers 32 32k;
            proxy_busy_buffers_size 128k;
            proxy_connect_timeout 60;
            proxy_read_timeout 120;
            proxy_send_timeout 60;

            proxy_pass_request_headers on;
            proxy_set_header referer $1://$proxy_host;

            #proxy_redirect off;
            proxy_pass $cookie_CORS/$1;
            proxy_redirect (.*) /;
            #proxy_redirect off;
            # add_header "Set-Cookie" "CORS=$1:/$proxy_host; HttpOnly; Secure;Path=/";
            add_header 'Access-Control-Allow-Origin' "$http_origin";
            add_header 'Access-Control-Allow-Methods' 'GET, POST, OPTIONS';
            add_header 'Access-Control-Allow-Headers' 'Origin, X-Requested-With, Content-Type, Accept';
        }

        error_page 500 502 503 504 /50x.html;
        location = /50x.html {
            root html;
        }
}
